def self.record(transaction, operations)
  operations.each do |operation|
    params = {
      block_transaction: transaction,
      type: "Bucket::Operation::#{operation.first.classify}",
      payload: operation.last.to_json,
    }
    
    Operation.create(params)
  end
end
